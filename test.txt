: O \ ( pin -- ) set pinMode as OUTPUT
  1 swap pinMode ;
: I \ ( pin -- ) set pinMode as INPUT
  2 swap pinMode ;
: H \ ( pin -- ) set pinLevel as HIGH
  1 swap pinWrite ;
: L \ ( pin -- ) set pinLevel as LOW
  0 swap pinWrite ;

: T \ ( pin -- ) set pinLevel as toggle
  1 over pinRead - swap pinWrite ;
: X \ ( n -- ) toggle led n times
  13 O 0 do 13 T 500 delay loop ;

: t \ ( -- ) show 9 * 9 table
  10 1 do cr 10 1 do i j * dup 10 < if space then loop loop ;
 
: y \ ( n -- ) check if n is zero
  if ." non-zero" else ." zero" then ;
: z \ ( n -- ) check if n is zero
  if ." non-" then ." zero" ;

: countRomWords ( -- n ) 
  1
  begin 1 + dup romEntry @ 0 =
  until 1 - ;
  
: romWords \ ( -- ) show all words defined in rom
  countRomWords
  begin dup while dup .name space 1 -
  repeat drop ;

: >link ( ramWordEntry -- lfa ) ;
: >cfa ( ramWordEntry -- cfa ) 4 + @ ;
: countRamWords ( -- n )
  0 ramLast
  begin dup
  while swap 1 + swap >link @
  repeat drop ;

: ramWords \ ( -- ) show all words defined in ram
  ramLast
  begin dup
  while dup >cfa .name space >link @
  repeat drop ;
  
: words \ ( -- ) show all words defined in ram and rom
  cr ramWords romWords ;

: seeRamWords \ ( -- ) see all words in ram
  countRomWords countRamWords +
  ramLast
  begin dup
  while swap dup . 1 - swap dup >cfa (see) >link @
  repeat 2drop ;

: seeRomWords \ ( -- ) see all words in rom
  countRomWords
  begin dup while dup . dup (see) 1 -
  repeat drop ;

: seeAll \ ( -- ) see all words
  seeRamWords seeRomWords ;
